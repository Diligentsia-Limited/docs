---
title: 'Create Company with InvestorMatch'
api: 'POST https://devserver.diligentsia.co.uk/api/v1/company/create-with-investormatch'
description: 'Register a UK company with comprehensive investor matching data'
---

<Note>
  This endpoint requires authentication via X-API-Key header. Contact support@diligentsia.com to obtain your partner API key.
</Note>

## Headers

<ParamField header="X-API-Key" type="string" required>
  Your partner API key for authentication
</ParamField>

<ParamField header="Content-Type" type="string" required>
  Must be `application/json`
</ParamField>

## Request Body

### User Information (Optional)

<ParamField body="user" type="object">
  <Expandable title="User fields">
    <ParamField body="email" type="string">
      User's email address. If provided, user will receive activation link.
    </ParamField>
    
    <ParamField body="firstName" type="string">
      User's first name
    </ParamField>
    
    <ParamField body="lastName" type="string">
      User's last name
    </ParamField>
  </Expandable>
</ParamField>

### Company Information (Required)

<ParamField body="company" type="object" required>
  <Expandable title="Company fields">
    <ParamField body="ukRegNo" type="string" required>
      UK company registration number (6-8 alphanumeric characters)
      
      **Pattern:** `^[0-9A-Z]{6,8}$`
      
      **Auto-transform:** Spaces removed, converted to uppercase
      
      **Examples:** `12345678`, `AB123456`, `SC123456`
    </ParamField>
    
    <ParamField body="companyWebsite" type="string" required>
      Company website URL
      
      **Format:** Valid URL (protocol optional)
      
      **Examples:** `example.com`, `https://example.co.uk`
    </ParamField>
    
    <ParamField body="company5Words" type="string" required>
      Company description in minimum 5 words
      
      **Example:** `Software development consulting services company`
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="status" type="number">
  HTTP status code (200 for success)
</ResponseField>

<ResponseField name="message" type="string">
  Response message
</ResponseField>

<ResponseField name="data" type="object">
  <Expandable title="Response data">
    <ResponseField name="companyId" type="string">
      Unique identifier for the created company
    </ResponseField>
    
    <ResponseField name="userId" type="string">
      Unique identifier for the user
    </ResponseField>
    
    <ResponseField name="companyName" type="string">
      Company name from Companies House
    </ResponseField>
    
    <ResponseField name="ukRegNo" type="string">
      Normalized UK registration number
    </ResponseField>
    
    <ResponseField name="partnerId" type="string">
      Partner organization ID
    </ResponseField>
    
    <ResponseField name="partnerName" type="string">
      Partner organization name
    </ResponseField>
    
    <ResponseField name="investorMatchCompleted" type="boolean">
      Whether InvestorMatch data was successfully saved
    </ResponseField>
    
    <ResponseField name="fitReport" type="object">
      <Expandable title="Fit report details">
        <ResponseField name="topMatches" type="array">
          Array of top 10 investor matches with scores
        </ResponseField>
        
        <ResponseField name="averageFitScore" type="number">
          Average compatibility score across all investors
        </ResponseField>
        
        <ResponseField name="companyProfile" type="object">
          Summary of company's InvestorMatch profile
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
